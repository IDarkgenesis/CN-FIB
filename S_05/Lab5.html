Potencies mod max

clear
A = [1 0 -1; 1 2 1; 2 2 3];
x= [1 1 1]';
n=50;
for i= 1:n
xaux= x;
z= A*x;
m= norm(z,'inf');
x= (1/m)*z;
if norm(x-xaux, 'inf') < eps
break;
end
end

clear

n=50;

%z=B*x;
z=A\x;
xaux= x;
m= norm(z,'inf');
x= (1/m) * z;
if norm(x - xaux) < eps
break;
end
end
